## List All Products

<api method="GET" endpoint="/api/v2/products" auth="none">
  Get paginated list of all products
</api>

### Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number
</ParamField>

### Example

```bash
curl https://your-domain.com/api/v2/products?page=1
```

## Search Products

<api method="GET" endpoint="/api/v2/products/search" auth="none">
  Search products with filters
</api>

### Query Parameters

<ParamField query="name" type="string">
  Product name or keywords to search
</ParamField>

<ParamField query="categories" type="string">
  Comma-separated category IDs (e.g., "1,2,3")
</ParamField>

<ParamField query="brands" type="string">
  Comma-separated brand IDs (e.g., "1,5,10")
</ParamField>

<ParamField query="min" type="number">
  Minimum price
</ParamField>

<ParamField query="max" type="number">
  Maximum price
</ParamField>

<ParamField query="sort_key" type="string">
  Sort order: `price_low_to_high`, `price_high_to_low`, `new_arrival`, `popularity`, `top_rated`
</ParamField>

### Example

```bash
curl "https://your-domain.com/api/v2/products/search?name=laptop&min=500&max=2000&sort_key=price_low_to_high"
```

## Get Variant Price

<api method="POST" endpoint="/api/v2/products/variant/price" auth="none">
  Get price for specific product variant
</api>

### Request

<ParamField body="id" type="integer" required>
  Product ID
</ParamField>

<ParamField body="quantity" type="integer" default="1">
  Quantity
</ParamField>

<ParamField body="color" type="string">
  Color code (without #)
</ParamField>

<ParamField body="variants" type="string">
  Comma-separated variant options
</ParamField>

### Response

<ResponseField name="result" type="boolean">
  Success status
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable>
    <ResponseField name="price" type="string">
      Formatted price with quantity
    </ResponseField>
    <ResponseField name="stock" type="integer">
      Available stock quantity
    </ResponseField>
    <ResponseField name="stock_txt" type="string">
      Stock status text
    </ResponseField>
    <ResponseField name="variant" type="string">
      Variant string
    </ResponseField>
    <ResponseField name="max_limit" type="integer">
      Maximum order quantity
    </ResponseField>
    <ResponseField name="in_stock" type="integer">
      In stock status (1 or 0)
    </ResponseField>
    <ResponseField name="image" type="string">
      Variant image URL
    </ResponseField>
  </Expandable>
</ResponseField>
